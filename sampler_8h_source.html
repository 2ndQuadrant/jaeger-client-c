152 static inline jaeger_sampler*
153 jaeger_sampler_choice_get_sampler(jaeger_sampler_choice* sampler)
154 {
155 #define JAEGERTRACINGC_SAMPLER_TYPE_CASE(type)
156  case jaeger_##type##_sampler_type:
157  return (jaeger_sampler*) &sampler->type##_sampler
158 
159  switch (sampler->type) {
160  JAEGERTRACINGC_SAMPLER_TYPE_CASE(const);
161  JAEGERTRACINGC_SAMPLER_TYPE_CASE(probabilistic);
162  JAEGERTRACINGC_SAMPLER_TYPE_CASE(rate_limiting);
163  JAEGERTRACINGC_SAMPLER_TYPE_CASE(guaranteed_throughput_probabilistic);
164  JAEGERTRACINGC_SAMPLER_TYPE_CASE(adaptive);
165  default:
166  jaeger_log_warn(
167  "Invalid sampler type in sampler choice, sampler type = %d",
168  sampler->type);
169  return NULL;
170  }
171 
172 #undef JAEGERTRACINGC_SAMPLER_TYPE_CASE
173 }

175 static inline void jaeger_sampler_choice_destroy(jaeger_sampler_choice* sampler)
176 {
177  assert(sampler != NULL);
178  jaeger_sampler* s = jaeger_sampler_choice_get_sampler(sampler);
179  if (s != NULL) {
180  ((jaeger_destructible*) s)->destroy((jaeger_destructible*) s);
181  }
