136 static inline bool jaeger_hashtable_rehash(jaeger_hashtable* hashtable)
137 {
138  assert(hashtable != NULL);
139  const size_t new_bucket_count = (1u << (hashtable->order + 1));
140  jaeger_list* new_buckets =
141  jaeger_malloc(new_bucket_count * sizeof(jaeger_list));
142  if (new_buckets == NULL) {
143  return false;
144  }
145  for (size_t i = 0; i < new_bucket_count; i++) {
146  new_buckets[i] = (jaeger_list) JAEGERTRACINGC_LIST_INIT;
147  }
148 
149  for (size_t i = 0; i < (1u << (hashtable->order)); i++) {
150  jaeger_list* bucket = &hashtable->buckets[i];
151  for (jaeger_list_node *entry = bucket->head, *next_entry; entry != NULL;
152  entry = next_entry) { /* Store next node before modifying current
153  node. */
154  next_entry = entry->next;
155  /* Remove entry, without deleting it, and reinsert into new bucket.
156  */
157  jaeger_list_node_remove(bucket, entry);
158  const size_t hash_code = jaeger_hashtable_hash(
159  ((jaeger_key_value_node*) entry)->data.key);
160  const size_t index = hash_code & (new_bucket_count - 1);
161  jaeger_list_append(&new_buckets[index], entry);
162  }
163  }
164 
165  jaeger_free(hashtable->buckets);
166  hashtable->buckets = new_buckets;
167  hashtable->order++;
168  return true;
169 }

176 static inline jaeger_hashtable_lookup_result
177 jaeger_hashtable_internal_lookup(jaeger_hashtable* hashtable, const char* key)
178 {
179  assert(hashtable != NULL);
180  assert(key != NULL);
181 
182  jaeger_hashtable_lookup_result result = {.node = NULL, .bucket = NULL};
183  const size_t bucket_count = (1u << hashtable->order);
184  const size_t hash_code = jaeger_hashtable_hash(key);
185  const size_t index = hash_code & (bucket_count - 1);
186  result.bucket = &hashtable->buckets[index];
187  for (jaeger_list_node* node = result.bucket->head; node != NULL;
188  node = node->next) {
189  if (strcmp(((jaeger_key_value_node*) node)->data.key, key) == 0) {
190  result.node = (jaeger_key_value_node*) node;
191  break;
192  }
193  }
194  return result;
195 }

208 static inline bool jaeger_hashtable_put(jaeger_hashtable* hashtable,
209  const char* key,
210  const char* value)
211 {
212  assert(hashtable != NULL);
213  assert(key != NULL);
214  assert(value != NULL);
215  const size_t bucket_count = (1u << hashtable->order);
216  if (((double) hashtable->size + 1) / bucket_count >=
217  JAEGERTRACINGC_HASHTABLE_THRESHOLD &&
218  !jaeger_hashtable_rehash(hashtable)) {
219  return false;
220  }
221 
222  jaeger_hashtable_lookup_result result =
223  jaeger_hashtable_internal_lookup(hashtable, key);
224  jaeger_key_value_node* entry = result.node;
225  jaeger_list* bucket = result.bucket;
226  if (entry != NULL) {
227  char* value_copy = jaeger_strdup(value);
228  if (value_copy == NULL) {
229  return false;
230  }
231  jaeger_free(entry->data.value);
232  entry->data.value = value_copy;
233  return true;
234  }
235 
236  jaeger_key_value kv;
237  if (!jaeger_key_value_init(&kv, key, value) ||
238  (entry = jaeger_key_value_node_new(kv)) == NULL) {
239  jaeger_key_value_destroy(&kv);
240  return false;
241  }
242  jaeger_list_insert(bucket, bucket->size, (jaeger_list_node*) entry);
243  hashtable->size++;
244 
245  return true;
246 }

249  const char* key)
250 {
251  jaeger_hashtable_lookup_result result =
252  jaeger_hashtable_internal_lookup(hashtable, key);
253  if (result.bucket != NULL && result.node != NULL) {
254  jaeger_list_node_remove(result.bucket, (jaeger_list_node*) result.node);

256  }

261 static inline bool jaeger_hashtable_copy(jaeger_hashtable* restrict dst,
262  const jaeger_hashtable* restrict src)
263 {
264  /* Avoid malloc of zero size. */
265  if (src->size == 0) {
266  return jaeger_hashtable_init(dst);
267  }
268 
269  *dst = (jaeger_hashtable) JAEGERTRACINGC_HASHTABLE_INIT;
270  const size_t order = jaeger_hashtable_minimal_order(src->size);
271  const size_t bucket_count = (1u << order);
272  dst->buckets = jaeger_malloc(bucket_count * sizeof(jaeger_list));
273  if (dst->buckets == NULL) {
274  goto cleanup;
275  }
276  dst->order = order;
277  for (size_t i = 0; i < bucket_count; i++) {
278  dst->buckets[i] = (jaeger_list) JAEGERTRACINGC_LIST_INIT;
279  }
280  for (size_t i = 0, len = (1u << src->order); i < len; i++) {
281  for (jaeger_list_node* entry = src->buckets[i].head; entry != NULL;
282  entry = entry->next) {
283  const jaeger_key_value* kv =
284  &((jaeger_key_value_node*) entry)->data;
285  if (!jaeger_hashtable_put(dst, kv->key, kv->value)) {
286  goto cleanup;
287  }
288  }
289  }
290 
291  assert(dst->size == src->size);
292  return true;
293 
294 cleanup:
295  jaeger_hashtable_destroy(dst);
296  return false;
297 }
