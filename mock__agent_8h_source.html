86 static inline bool
87 read_client_request(int client_fd, char* buffer, int* buffer_len)
88 {
89  int num_read = read(client_fd,
90  buffer,
91  JAEGERTRACINGC_HTTP_SAMPLING_MANAGER_REQUEST_MAX_LEN);
92 
93  while (num_read > 0) {
94  TEST_ASSERT_LESS_OR_EQUAL(
95  JAEGERTRACINGC_HTTP_SAMPLING_MANAGER_REQUEST_MAX_LEN, *buffer_len);
96  TEST_ASSERT_LESS_OR_EQUAL(
97  JAEGERTRACINGC_HTTP_SAMPLING_MANAGER_REQUEST_MAX_LEN,
98  *buffer_len + num_read);
99  *buffer_len += num_read;
100  if (*buffer_len >= 4 &&
101  memcmp(&buffer[*buffer_len - 4], "\r\n\r\n", 4) == 0) {
102  break;
103  }
104  num_read = read(client_fd,
105  buffer,
106  JAEGERTRACINGC_HTTP_SAMPLING_MANAGER_REQUEST_MAX_LEN);
107  }
108 
109  return (num_read >= 0);
110 }

122 static inline void* mock_http_server_run_loop(void* context)
123 {
124  TEST_ASSERT_NOT_NULL(context);
125  mock_http_server_context* server_context =
126  (mock_http_server_context*) context;
127  mock_http_server* server = server_context->server;
128  TEST_ASSERT_NOT_NULL(server);
129  server->client_fd = -1;
130 
131  jaeger_mutex_lock(&server->mutex);
132  server->running = true;
133  jaeger_mutex_unlock(&server->mutex);
134  jaeger_cond_signal(&server_context->cv);
135 
136  const int client_fd = accept(server->server_fd, NULL, 0);
137  TEST_ASSERT_GREATER_OR_EQUAL(0, client_fd);
138 
139  jaeger_mutex_lock(&server->mutex);
140  server->client_fd = client_fd;
141  jaeger_mutex_unlock(&server->mutex);
142 
143  while (true) {
144  http_parser parser;
145  http_parser_init(&parser, HTTP_REQUEST);
146  parser.data = server;
147  http_parser_settings settings;
148  memset(&settings, 0, sizeof(settings));
149  settings.on_url = &mock_http_server_on_url;
150 
151  server->url_len = 0;
152  char buffer[JAEGERTRACINGC_HTTP_SAMPLING_MANAGER_REQUEST_MAX_LEN] = {
153  '\0'};
154  int buffer_len = 0;
155  if (!read_client_request(server->client_fd, buffer, &buffer_len) ||
156  buffer_len == 0) {
157  break;
158  }
159 
160  const int parsed_len =
161  http_parser_execute(&parser, &settings, buffer, buffer_len);
162  TEST_ASSERT_EQUAL(buffer_len, parsed_len);
163  struct http_parser_url url;
164  TEST_ASSERT_EQUAL(
165  0,
166  http_parser_parse_url(
167  &server->url_buffer[0], server->url_len, false, &url));
168  TEST_ASSERT_TRUE(url.field_set & (1 << UF_QUERY));
169  char query[url.field_data[UF_QUERY].len + 1];
170  memcpy(query,
171  &server->url_buffer[url.field_data[UF_QUERY].off],
172  sizeof(query) - 1);
173  query[sizeof(query) - 1] = '\0';
174 
175  const mock_http_response* response = NULL;
176  jaeger_mutex_lock(&server->mutex);
177  for (int i = 0, len = jaeger_vector_length(&server->responses); i < len;
178  i++) {
179  const mock_http_response* existing_response =
180  (const mock_http_response*) jaeger_vector_offset(
181  &server->responses, i);
182  TEST_ASSERT_NOT_NULL(existing_response);
183  TEST_ASSERT_NOT_NULL(existing_response->service_name);
184  char buffer[strlen("service=") +
185  strlen(existing_response->service_name) + 1];
186  TEST_ASSERT_EQUAL(sizeof(buffer) - 1,
187  snprintf(buffer,
188  sizeof(buffer),
189  "service=%s",
190  existing_response->service_name));
191  if (strcmp(buffer, query) == 0) {
192  response = existing_response;
193  break;
194  }
195  }
196  TEST_ASSERT_NOT_NULL(response);
197 
198  char strategy_response[strlen(response->json_format) + 10];
199  TEST_ASSERT_LESS_THAN(sizeof(strategy_response),
200  snprintf(strategy_response,
201  sizeof(strategy_response),
202  response->json_format,
203  response->arg_value));
204  const char http_response_format[] = "HTTP/1.1 200 OK\r\n"
205  "Content-Type: application/json\r\n"
206  "Content-Length: %zu\r\n\r\n"
207  "%s";
208  const int num_digits = 10;
209  char http_response[strlen(strategy_response) +
210  strlen(http_response_format) + num_digits + 1];
211  const int result = snprintf(http_response,
212  sizeof(http_response),
213  http_response_format,
214  strlen(strategy_response),
215  strategy_response);
216  TEST_ASSERT_LESS_OR_EQUAL(sizeof(http_response) - 1, result);
217  jaeger_mutex_unlock(&server->mutex);
218 
219  const int num_written =
220  write(server->client_fd, http_response, strlen(http_response));
221  TEST_ASSERT_EQUAL(strlen(http_response), num_written);
222  }
223  return NULL;
224 }

257  jaeger_mutex_lock(&server->mutex);
258  mock_http_response* response_copy = NULL;
259  for (int i = 0, len = jaeger_vector_length(&server->responses); i < len;
260  i++) {
261  mock_http_response* existing_response =
262  (mock_http_response*) jaeger_vector_offset(&server->responses, i);
263  TEST_ASSERT_NOT_NULL(existing_response);
264  TEST_ASSERT_NOT_NULL(existing_response->service_name);
265  if (strcmp(existing_response->service_name, response->service_name) ==
266  0) {
267  mock_http_response_destroy(existing_response);
268  response_copy = existing_response;
269  }
270  }
271  if (response_copy == NULL) {
272  response_copy = jaeger_vector_append(&server->responses);
273  }
274  TEST_ASSERT_NOT_NULL(response_copy);
275  mock_http_response_copy(response_copy, response);
276  jaeger_mutex_unlock(&server->mutex);
277 }

279 static inline void mock_http_server_start(mock_http_server* server)
280 {
281  TEST_ASSERT_NOT_NULL(server);
282 
283  TEST_ASSERT_TRUE(
284  jaeger_vector_init(&server->responses, sizeof(mock_http_response)));
285 
286  server->server_fd = socket(AF_INET, SOCK_STREAM, 0);
287  TEST_ASSERT_GREATER_OR_EQUAL(0, server->server_fd);
288  server->addr.sin_addr.s_addr = htonl(INADDR_ANY);
289  server->addr.sin_family = AF_INET;
290  server->addr.sin_port = 0;
291  TEST_ASSERT_EQUAL(0,
292  bind(server->server_fd,
293  (struct sockaddr*) &server->addr,
294  sizeof(server->addr)));
295  TEST_ASSERT_EQUAL(0, listen(server->server_fd, MOCK_HTTP_BACKLOG));
296  socklen_t addr_len = sizeof(server->addr);
297  TEST_ASSERT_EQUAL(0,
298  getsockname(server->server_fd,
299  (struct sockaddr*) &server->addr,
300  &addr_len));
301  TEST_ASSERT_EQUAL(sizeof(server->addr), addr_len);
302 
303  mock_http_server_context context = MOCK_HTTP_SERVER_CONTEXT_INIT;
304  context.server = server;
305 
306  jaeger_mutex_lock(&server->mutex);
307  TEST_ASSERT_EQUAL(0,
308  jaeger_thread_init(&server->thread,
309  &mock_http_server_run_loop,
310  &context));
311  while (!server->running) {
312  jaeger_cond_wait(&context.cv, &server->mutex);
313  }
314  jaeger_mutex_unlock(&server->mutex);
315  jaeger_cond_destroy(&context.cv);
316 }

318 static inline void mock_http_server_destroy(mock_http_server* server)
319 {
320  jaeger_mutex_lock(&server->mutex);
321  if (server->client_fd > -1) {
322  shutdown(server->client_fd, SHUT_RDWR);
323  close(server->client_fd);
324  server->client_fd = -1;
325  }
326  if (server->server_fd > -1) {
327  shutdown(server->server_fd, SHUT_RDWR);
328  close(server->server_fd);
329  server->server_fd = -1;
330  }
331  if (server->thread != 0) {
332  server->running = false;
333  jaeger_thread_join(server->thread, NULL);
334  server->thread = 0;
335  }
336  jaeger_mutex_unlock(&server->mutex);
337  jaeger_mutex_destroy(&server->mutex);
338  memset(&server->addr, 0, sizeof(server->addr));
339  JAEGERTRACINGC_VECTOR_FOR_EACH(
340  &server->responses, mock_http_response_destroy, mock_http_response);
341  jaeger_vector_destroy(&server->responses);
342 }
