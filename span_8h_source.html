111 static inline void jaeger_span_context_destroy(jaeger_destructible* d)
112 {
113  if (d == NULL) {
114  return;
115  }
116  jaeger_span_context* ctx = (jaeger_span_context*) d;
117  jaeger_hashtable_destroy(&ctx->baggage);
118  if (ctx->debug_id != NULL) {
119  jaeger_free(ctx->debug_id);
120  ctx->debug_id = NULL;
121  }
122  jaeger_mutex_destroy(&ctx->mutex);
123 }

125 static inline void jaeger_span_context_foreach_baggage_item(
126  opentracing_span_context* span_context,
127  opentracing_bool (*f)(void* arg, const char* key, const char* value),
128  void* arg)
129 {
130  jaeger_span_context* ctx = (jaeger_span_context*) span_context;
131  jaeger_mutex_lock(&ctx->mutex);
132 
133  jaeger_hashtable* baggage = &((jaeger_span_context*) span_context)->baggage;
134  for (size_t i = 0, len = 1u << baggage->order; i < len; i++) {
135  for (const jaeger_list_node* node = baggage->buckets[i].head;
136  node != NULL;
137  node = node->next) {
138  const jaeger_key_value* kv = &((jaeger_key_value_node*) node)->data;
139  if (!f(arg, kv->key, kv->value)) {
140  break;
141  }
142  }
143  }
144 
145  jaeger_mutex_unlock(&ctx->mutex);
146 }

155 static inline bool
156 jaeger_span_context_copy(jaeger_span_context* restrict dst,
157  const jaeger_span_context* restrict src)
158 {
159  assert(dst != NULL);
160  assert(src != NULL);
161  *dst = (jaeger_span_context) JAEGERTRACINGC_SPAN_CONTEXT_INIT;
162  jaeger_lock((jaeger_mutex*) &src->mutex, &dst->mutex);
163  if (!jaeger_hashtable_copy(&dst->baggage, &src->baggage)) {
164  jaeger_mutex_unlock((jaeger_mutex*) &src->mutex);
165  jaeger_mutex_unlock(&dst->mutex);
166  jaeger_span_context_destroy((jaeger_destructible*) dst);
167  return false;
168  }
169  dst->trace_id = src->trace_id;
170  dst->span_id = src->span_id;
171  dst->parent_id = src->parent_id;
172  dst->flags = src->flags;
173  jaeger_mutex_unlock((jaeger_mutex*) &src->mutex);
174  jaeger_mutex_unlock(&dst->mutex);
175  return true;
176 }

233 static inline bool
234 jaeger_span_ref_copy(void* arg, void* restrict dst, const void* restrict src)
235 {
236  (void) arg;
237  assert(dst != NULL);
238  assert(src != NULL);
239  jaeger_span_ref* dst_span_ref = (jaeger_span_ref*) dst;
240  /* Do not call jaeger_span_ref_init so we avoid memory leak in
241  * jaeger_span_context_copy due to double initialization of span
242  * context.

245  if (!jaeger_span_context_copy(&dst_span_ref->context,
246  &src_span_ref->context)) {
247  return false;
248  }
249  dst_span_ref->type = src_span_ref->type;
250  return true;
251 }

413 /**
414  * Finish span with options.
415  * @param span The span instance.
416  * @param options Options to determine finish timestamp, etc. May be NULL.
417  */
418 static inline void
419 jaeger_span_finish_with_options(opentracing_span* s,
420  const opentracing_finish_span_options* options)
421 {
422  assert(s != NULL);
423  jaeger_span* span = (jaeger_span*) s;
424  if (options == NULL) {
425  opentracing_finish_span_options default_finish_options =
426  JAEGERTRACINGC_FINISH_SPAN_OPTIONS_INIT;
427  jaeger_span_finish_with_options(s, &default_finish_options);
428  return;
429  }
430  jaeger_mutex_lock(&span->mutex);
431  if (jaeger_span_is_sampled_no_locking(span)) {
432  jaeger_duration finish_time = options->finish_time;
433  if (finish_time.value.tv_sec == 0 && finish_time.value.tv_nsec == 0) {
434  jaeger_duration_now(&finish_time);
435  }
436  const bool result = jaeger_time_subtract(finish_time.value,
437  span->start_time_steady.value,
438  &span->duration.value);
439  (void) result;
440  assert(result);
441 
442  assert(options->num_log_records == 0 || options->log_records != NULL);
443  for (int i = 0; i < options->num_log_records; i++) {
444  jaeger_span_log_no_locking(span, &options->log_records[i]);
445  }
446  }
447  jaeger_mutex_unlock(&span->mutex);
448 
449  /* Call jaeger_tracer_report_span even for non-sampled traces, in case
450  * we need to return the span to a pool.
451  */
452  jaeger_tracer_report_span(span->tracer, span);
453 }

571 /**
572  * Set sampling flag on span.
573  * @internal
574  * @param span The span instance.
575  * @param tag The tag value to consider for sampling.
576  * @return True on success, false otherwise.
577  */
578 static inline bool
579 jaeger_span_set_sampling_priority(jaeger_span* span,
580  const opentracing_value* value)
581 {
582  assert(span != NULL);
583  assert(value != NULL);
584  switch (value->type) {
585  case opentracing_value_int64:
586  break;
587  case opentracing_value_uint64:
588  break;
589  default:
590  return false;
591  }
592 
593  bool success = false;
594  jaeger_lock(&span->mutex, &span->context.mutex);
595  if ((value->type == opentracing_value_int64 && value->value.int64_value) ||
596  (value->type == opentracing_value_uint64 &&
597  value->value.uint64_value)) {
598  span->context.flags |= ((uint8_t) jaeger_sampling_flag_debug) |
599  ((uint8_t) jaeger_sampling_flag_sampled);
600  success = true;
601  }
602  else {
603  span->context.flags &= (~((uint8_t) jaeger_sampling_flag_sampled));
604  }
605  jaeger_mutex_unlock(&span->mutex);
606  jaeger_mutex_unlock(&span->context.mutex);
607  return success;
608 }

906 static inline bool jaeger_span_context_scan(jaeger_span_context* ctx,
907  const char* str)
908 {
909  assert(ctx != NULL);
910  assert(str != NULL);
911  jaeger_trace_id trace_id = JAEGERTRACINGC_TRACE_ID_INIT;
912  const int len = strlen(str);
913  char buffer[len + 1];
914  memcpy(buffer, str, len);
915  buffer[len] = '\0';
916  uint64_t span_id = 0;
917  uint64_t parent_id = 0;
918  uint8_t flags = 0;
919  char* token = buffer;
920  char* token_context = NULL;
921  char* token_end = NULL;
922  for (int i = 0; i < 5; i++) {
923  token = strtok_r(i == 0 ? token : NULL, ":", &token_context);
924  if (token == NULL && i != 4) {
925  return false;
926  }
927  token_end = NULL;
928  switch (i) {
929  case 0:
930  if (!jaeger_trace_id_scan(&trace_id, token)) {
931  return false;
932  }
933  break;
934  case 1:
935  span_id = strtoull(token, &token_end, 16);
936  break;
937  case 2:
938  parent_id = strtoull(token, &token_end, 16);
939  break;
940  case 3:
941  flags = strtoull(token, &token_end, 16);
942  break;
943  default:
944  assert(i == 4);
945  break;
946  }
947  if (token_end != NULL && *token_end != '\0') {
948  return false;
949  }
950  }
951 
952  jaeger_mutex_lock(&ctx->mutex);
953  ctx->trace_id = trace_id;
954  ctx->span_id = span_id;
955  ctx->parent_id = parent_id;
956  ctx->flags = flags;
957  jaeger_mutex_unlock(&ctx->mutex);
958  return true;
959 }
