86 // Algorithm based on
87 // http://www.gnu.org/software/libc/manual/html_node/Elapsed-Time.html.
88 static inline bool jaeger_time_subtract(opentracing_time_value lhs,
89  opentracing_time_value rhs,
90  opentracing_time_value* result)
91 {
92  assert(result != NULL);
93 
94  if (lhs.tv_nsec < rhs.tv_nsec) {
95  const int64_t nsec = (rhs.tv_nsec - lhs.tv_nsec) /
96  JAEGERTRACINGC_NANOSECONDS_PER_SECOND +
97  1;
98  rhs.tv_nsec -= JAEGERTRACINGC_NANOSECONDS_PER_SECOND * nsec;
99  rhs.tv_sec += nsec;
100  }
101 
102  if (lhs.tv_nsec - rhs.tv_nsec > JAEGERTRACINGC_NANOSECONDS_PER_SECOND) {
103  const int64_t nsec =
104  (lhs.tv_nsec - rhs.tv_nsec) / JAEGERTRACINGC_NANOSECONDS_PER_SECOND;
105  rhs.tv_nsec += JAEGERTRACINGC_NANOSECONDS_PER_SECOND * nsec;
106  rhs.tv_sec -= nsec;
107  }
108 
109  result->tv_sec = lhs.tv_sec - rhs.tv_sec;
110  result->tv_nsec = lhs.tv_nsec - rhs.tv_nsec;
111  return lhs.tv_sec >= rhs.tv_sec;
112 }
