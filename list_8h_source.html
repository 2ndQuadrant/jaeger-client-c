56 static inline jaeger_list_node* jaeger_list_get(jaeger_list* list, size_t index)
57 {
58  assert(list != NULL);
59  if (index >= list->size) {
60  return NULL;
61  }
62 
63  jaeger_list_node* node = list->head;
64  for (size_t i = 0; i < index; i++) {
65  node = node->next;
66  }
67  return node;
68 }

70 static inline void
71 jaeger_list_insert(jaeger_list* list, size_t index, jaeger_list_node* elem)
72 {
73  assert(list != NULL);
74  assert(elem != NULL);
75  assert(index <= list->size);
76 
77  if (index == 0) {
78  if (list->head == NULL) {
79  elem->next = NULL;
80  elem->prev = NULL;
81  }
82  else {
83  assert(list->head->prev == NULL);
84  elem->prev = NULL;
85  elem->next = list->head;
86  list->head->prev = elem;
87  }
88  list->head = elem;
89  list->size++;
90  return;
91  }
92 
93  assert(index > 0);
94  jaeger_list_node* node = jaeger_list_get(list, index - 1);
95  assert(node != NULL);
96  elem->next = node->next;
97  if (elem->next != NULL) {
98  elem->next->prev = elem;
99  }
100  node->next = elem;
101  elem->prev = node;
102 
103  list->size++;
104 }

111 static inline void jaeger_list_node_remove(jaeger_list* list,
112  jaeger_list_node* node)
113 {
114  assert(list != NULL);
115  assert(node != NULL);
116 
117  if (list->head == node) {
118  if (node->next != NULL) {
119  list->head = node->next;
120  }
121  else {
122  list->head = NULL;
123  }
124  }
125 
126  if (node->next != NULL) {
127  node->next->prev = node->prev;
128  }
129  if (node->prev != NULL) {
130  node->prev->next = node->next;
131  }
132  node->next = NULL;
133  node->prev = NULL;
134 
135  list->size--;
136 }

138 static inline void jaeger_list_clear(jaeger_list* list)
139 {
140  if (list == NULL) {
141  return;
142  }
143 
144  for (jaeger_list_node *node = list->head, *next_node; node != NULL;
145  node = next_node) {
146  next_node = node->next;
147  ((jaeger_destructible*) node)->destroy((jaeger_destructible*) node);
148  }
149 
150  list->head = NULL;
151  list->size = 0;
152 }
