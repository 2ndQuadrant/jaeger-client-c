87 static inline void
88 jaeger_per_operation_strategy_destroy(jaeger_per_operation_strategy* strategy)
89 {
90  assert(strategy != NULL);
91  if (strategy->per_operation_strategy != NULL) {
92  for (int i = 0; i < (int) strategy->n_per_operation_strategy; i++) {
93  if (strategy->per_operation_strategy[i] != NULL) {
94  jaeger_operation_strategy_destroy(
95  strategy->per_operation_strategy[i]);
96  jaeger_free(strategy->per_operation_strategy[i]);
97  }
98  }
99  jaeger_free(strategy->per_operation_strategy);
100  strategy->per_operation_strategy = NULL;
101  }
102 }
